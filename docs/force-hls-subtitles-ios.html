Example Code:

To force subtitle in hls streem even if the manifest does not declares them you should add

        
            html5: {
              hls: {
                overrideNative: true,
                },
            }

to the player configuration and them add you text tracks in the source

Make sure to add

      
        crossorigin="anonymous"

to the video tag fie the subtitles are coming from a different origin

      
        <video
          id="player"
          controls
          muted
          class="cld-video-player"
          crossorigin="anonymous"
          width="500"
        ></video>
      
        var cld = cloudinary.Cloudinary.new({cloud_name: 'demo'});
        var player = cld.videoPlayer('player', {
          html5: {
            hls: {
              overrideNative: true,
            },
          },
        });

        player.source(
            'sea_turtle',
            {
              sourceTypes: ['hls'],
              transformation: {streaming_profile: 'hd'},
              textTracks: {
                captions: {
                  label: "English(captions)",
                  language: "en",
                  default: true,
                  url: "https://res.cloudinary.com/demo/raw/upload/outdoors.vtt"
                },
                subtitles: [
                  {
                    label: "English",
                    language: "en",
                    url: "https://res.cloudinary.com/demo/raw/upload/outdoors.vtt"
                  }
               ]
              }
            }
      );
